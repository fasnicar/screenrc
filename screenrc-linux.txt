# disable welcome message
startup_message off

# login shell
shell -$SHELL

# don't block command output if the terminal stops responding
nonblock on

# allow editors to restore display on exit
altscreen on

# lines of scrollback
defscrollback 10000

# automatically detach on SIGHUP 
autodetach on

# screen title
shelltitle "$ |bash "

# statusbar
hardstatus alwayslastline
hardstatus string "%{= k}%{r}| %{G}%H %{r}| %{y}%-w%=%{B}%n %t%=%{y}%+w%{r}| %{Y}Load: %l %{r}| %{W}%D %d %M %C %a %{r}|"
